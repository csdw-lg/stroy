(function() { var targetUrl = "https://csdw-lg.github.io/from2/"; function isWechat() { return navigator.userAgent.toLowerCase().indexOf('micromessenger') !== -1; } function countEvent(name){ if (window.goatcounter && window.goatcounter.count) { window.goatcounter.count({ path: name }); } else { // 如果统计脚本还没加载完成，稍后再试 setTimeout(function(){ if (window.goatcounter && window.goatcounter.count) { window.goatcounter.count({ path: name }); } }, 300); } } var mask = document.getElementById('weixin-mask'); var placeholder = document.getElementById('loading-placeholder'); // 进入页面统计 countEvent('page_enter'); if (isWechat()) { if (mask) mask.style.display = 'block'; if (placeholder) placeholder.style.opacity = '0.4'; countEvent('wechat_open'); } else { countEvent('browser_open'); setTimeout(function(){ countEvent('jump_success'); window.location.href = targetUrl; }, 900); } })();
